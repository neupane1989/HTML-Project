const GOOGLE_SHEETS_URL = "https://docs.google.com/spreadsheets/d/1aV9fmN_xkzaKTn-tKCXLS9-Vss_9Os9zkEtrVmoWFvE/export?format=csv"; // Extract query parameters const urlParams = new URLSearchParams(window.location.search); const area = urlParams.get('area'); const store = urlParams.get('store'); // Display area and store in the title document.title = `UAE Ramadan Performance - ${area} - ${store}`; // Fetch and process data from Google Sheets async function fetchPerformanceData() { try { const response = await fetch(GOOGLE_SHEETS_URL); if (!response.ok) throw new Error('Failed to fetch data'); const csvData = await response.text(); const workbook = XLSX.read(csvData, { type: 'string' }); const performanceSheet = workbook.Sheets[workbook.SheetNames[1]]; // Sheet2 for UAE Performance const performanceData = XLSX.utils.sheet_to_json(performanceSheet); // Filter data for the selected area and store const filteredData = performanceData.filter(row => row.Area === area && row['Location Name'] === store); // Populate the table with filtered data populatePerformanceTable(filteredData); } catch (error) { console.error('Error fetching performance data:', error); alert('Failed to load performance data. Please try again.'); } } // Populate the table with performance data function populatePerformanceTable(data) { const tbody = document.querySelector('#performanceTable tbody'); tbody.innerHTML = ''; // Clear existing rows data.forEach(row => { tbody.innerHTML += ` ${row.Area} ${row['Location Code']} ${row['Location Name']} ${row['Value_TY']} ${row['TGT_VALUE']} ${row['% TGT Ach']} ${row['Value_LY']} ${row['% Ach on LY']} `; }); } // Initialize the page fetchPerformanceData();
