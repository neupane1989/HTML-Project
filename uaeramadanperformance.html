function updateTable() { const area = document.getElementById('area').value; const store = document.getElementById('store').value; const filteredData = allData.filter(row => { return (area === 'All' || row.Area === area) && (store === 'All' || row.Store === store); }); const tbody = document.querySelector('#performanceTable tbody'); tbody.innerHTML = filteredData.map(row => ` ${row.Area} ${row['Store Code']} ${row.Store} ${row.Value_TY} ${row.TGT_VALUE} ${row['% TGT Ach']} ${row.Value_LY} ${row['% Ach LY']} `).join(''); }
