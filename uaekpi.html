const GOOGLE_SHEETS_URL = "https://docs.google.com/spreadsheets/d/1aV9fmN_xkzaKTn-tKCXLS9-Vss_9Os9zkEtrVmoWFvE/export?format=csv"; // Extract query parameters const urlParams = new URLSearchParams(window.location.search); const area = urlParams.get('area'); const store = urlParams.get('store'); // Display area and store in the title document.title = `UAE KPI's - ${area} - ${store}`; // Fetch and process data from Google Sheets async function fetchKPIData() { try { const response = await fetch(GOOGLE_SHEETS_URL); if (!response.ok) throw new Error('Failed to fetch data'); const csvData = await response.text(); const workbook = XLSX.read(csvData, { type: 'string' }); const kpiSheet = workbook.Sheets[workbook.SheetNames[3]]; // Sheet4 for KPI's const kpiData = XLSX.utils.sheet_to_json(kpiSheet); // Filter data for the selected area and store const filteredData = kpiData.filter(row => row.Area === area && row['Location Name'] === store); // Populate the table with filtered data populateKPITable(filteredData); } catch (error) { console.error('Error fetching KPI data:', error); alert('Failed to load KPI data. Please try again.'); } } // Populate the table with KPI data function populateKPITable(data) { const tbody = document.querySelector('#kpiTable tbody'); tbody.innerHTML = ''; // Clear existing rows data.forEach(row => { tbody.innerHTML += ` ${row.Area} ${row['Location Code']} ${row['Location Name']} ${row['ATV_TY']} ${row['UPT_TY']} ${row['ASP_TY']} ${row['SPD_TY']} ${row['QPD_TY']} ${row['S Care Value_TY']} ${row['% S Care Cont_TY']} ${row['Hosiery QTY_TY']} ${row['% Hosiery Cont_TY']} ${row['LHB Value_TY']} ${row['% LHB Cont_TY']} `; }); } // Initialize the page fetchKPIData();
