const GOOGLE_SHEETS_URL = "https://docs.google.com/spreadsheets/d/1aV9fmN_xkzaKTn-tKCXLS9-Vss_9Os9zkEtrVmoWFvE/export?format=csv"; // Extract query parameters const urlParams = new URLSearchParams(window.location.search); const area = urlParams.get('area'); const store = urlParams.get('store'); // Display area and store in the title document.title = `Roster - ${area} - ${store}`; // Function to generate dynamic date headers function generateDateHeaders() { const thead = document.querySelector('#rosterTable thead tr'); const today = new Date(); for (let i = 0; i < 6; i++) { const date = new Date(today); date.setDate(today.getDate() + i); const dateString = date.toLocaleDateString('en-GB', { day: '2-digit', month: 'short' }); thead.innerHTML += `${dateString}`; } } // Fetch and process data from Google Sheets async function fetchRotaData() { try { const response = await fetch(GOOGLE_SHEETS_URL); if (!response.ok) throw new Error('Failed to fetch data'); const csvData = await response.text(); const workbook = XLSX.read(csvData, { type: 'string' }); const rotaSheet = workbook.Sheets[workbook.SheetNames[2]]; // Sheet3 for Rota const rotaData = XLSX.utils.sheet_to_json(rotaSheet); // Filter data for the selected area and store const filteredData = rotaData.filter(row => row.Area === area && row.Store === store); // Populate the table with filtered data populateRosterTable(filteredData); } catch (error) { console.error('Error fetching rota data:', error); alert('Failed to load rota data. Please try again.'); } } // Populate the table with roster data function populateRosterTable(data) { const tbody = document.querySelector('#rosterTable tbody'); tbody.innerHTML = ''; // Clear existing rows data.forEach(row => { tbody.innerHTML += ` ${row.Store} ${row.HRMS} ${row.Designation} ${row['Staff Name']} ${row['Week No.']} ${['17-Feb-25', '18-Feb-25', '19-Feb-25', '20-Feb-25', '21-Feb-25', '22-Feb-25'] .map(date => `${row[date] || ''}`).join('')} ${row.Remarks || ''} `; }); } // Initialize the page generateDateHeaders(); fetchRotaData();
